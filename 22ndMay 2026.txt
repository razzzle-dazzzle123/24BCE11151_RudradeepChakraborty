//queries for today
use vitcampus;

create table departments(
    did int primary key,
    dname varchar(25) not null unique
);

insert into departments values
(101, "CSE"),
(102, "ECE"),
(103, "Civil"),
(106, "Aero"),
(107, "Mech");

select * from departments order by did;

create table hod(
    hid int primary key,
    hname varchar(20) not null unique
);

insert into hod values
(101, "Arun"),
(102, "Rudra"),
(104, "Karan"),
(105, "Meera"),
(107, "Soham");

select * from hod;

select *
from departments
cross join hod
order by did asc;

select *
from departments
inner join hod
on departments.did = hod.hid;

select *
from departments
natural join hod;

select *
from departments
left outer join hod
on departments.did = hod.hid
order by did asc;

select *
from departments
right outer join hod
on departments.did = hod.hid
order by did asc;

select *
from departments
left join hod
on departments.did = hod.hid

union

select *
from departments
right join hod
on departments.did = hod.hid;
