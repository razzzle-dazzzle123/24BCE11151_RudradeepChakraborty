1.List the Employee Name, their Department Name, and the Location Name for all employees.

Answer:
select employee_name, dept_name, location_id from employees join departments on employees.dept_id = departments.dept_id;


2.Retrieve the Project Name, Client Name, and the Total Hours Worked on that project, but only for projects managed by the 'Consulting' department.

Answer:
select project_name, client_name, sum(hours_worked) from projects join assignments on projects.project_id = assignments.project_id join departments
on projects.dept_id = departments.dept_id where dept_name = 'Consulting' group by project_name, client_name;


3.List all Departments that have no Employees currently assigned to them.

Answer:
select dept_name from departments where dept_id not in (select dept_id from employees);


4.Calculate the Average Hours Worked per Employee across all projects. List the Employee Name and the average hours.

Answer:
select employee_name,avg(hours_worked) from employees join assignments on employees.employee_id = assignments.employee_id group by employee_name;


5.Identify the Client Name(s) whose projects have a total combined hours worked exceeding 500 hours.

Answer:
select client_name from projects join assignments on projects.project_id = assignments.project_id group by client_name having sum(hours_worked) > 500;


6.Calculate the total number of employees assigned to each Project, but only include projects that have more than 5 employees assigned.

Answer:
select project_name,count(employee_id) as total_employees from projects join assignments on projects.project_id = assignments.project_id
group by project_name having count(employee_id) > 5;


7.Retrieve the Department Name(s) that have employees assigned to more than one project.

Answer:
select dept_name from departments join employees on departments.dept_id = employees.dept_id join assignments on employees.employee_id = assignments.employee_id
group by dept_name having count(distinct assignments.project_id) > 1;


8.Write a query to list the Employee Name and Project Name for any employee who has worked less than 10 hours on a project.

Answer:
select employee_name, project_name from employees join assignments on employees.employee_id = assignments.employee_id
join projects on assignments.project_id = projects.project_id where hours_worked < 10;


9.Find the Project Name(s) that have no assigned employees.

Answer:
select project_name from projects where project_id not in (select project_id from assignments);


10.Assuming the Assignments table has a start_date column, find the Employee Name and their oldest project start date.

Answer:
select employee_name, min(assignments.start_date) as oldestStartDate
from employees join assignments on employees.employee_id = assignments.employee_id group by employee_name;


11.Write a MySQL query using the DATE_FORMAT function to list all projects that were active in the year 2023.

Answer:
select project_name, start_date from projects where date_format(start_date,'%Y') = '2023';


12.Use the MySQL-specific IFNULL function to list all Employee Names. If an employee has no manager, display "CEO" instead of NULL.

Answer:
select employee_name,ifnull(cast(manager_id as char),'CEO') as manager from employees;


13.Find the top 3 Departments based on the total hours worked by all their employees across all projects.

Answer:
select dept_name,sum(hours_worked) as total_hours from departments join employees on departments.dept_id = employees.dept_id
join assignments on employees.employee_id = assignments.employee_id group by dept_nameorder by total_hours desc limit 3;


14.Display the Project Name and a status ("High Activity" if total hours > 300, "Normal" otherwise) using the CASE statement.

Answer:
select project_name,
case
    when sum(hours_worked) > 300 then 'High Activity'
    else 'Normal'
end as status
from projects join assignments on projects.project_id = assignments.project_id group by project_name;


15.List the Employee Name and their Manager's Name, even if the employee has no manager, or if a potential manager has no employees reporting to them.

Answer:
select employees.employee_name,managers.employee_name from employees left join employees managers on employees.manager_id = managers.employee_id;


16.Find the Employee Name(s) who are assigned to the same projects as Employee ID 101.

Answer:
select employee_name from employees join assignments on employees.employee_id = assignments.employee_id where assignments.project_id in
(select project_id from assignments where employee_id = 101) and employees.employee_id != 101;
